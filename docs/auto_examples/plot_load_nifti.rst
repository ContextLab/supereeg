    # import
    import superEEG as se
    from nilearn import plotting as ni_plt

    # load nifti -> brain object
    bo = se.load('example_nifti')

    # export brain object -> nifti
    nifti = bo.to_nii()

    # plot the result
    ni_plt.plot_anat(nifti)
    ni_plt.show()
